## Usage

```bash
# install dependencies
$ npm install buyi-ui --save
```

```jsx | tsx
import React from 'react';
import ReactDOM from 'react-dom';
import { Button } from 'buyi-ui';

ReactDOM.render(<Button>Button</Button>, document.getElementById('root'));
```
